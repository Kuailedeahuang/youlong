## 三、核心算法规格

### 3.1 每日自然价格波动算法

```javascript
function calculateNaturalFluctuation(item) {
    const { min, max } = item.naturalFluctuation;
    const fluctuation = randomRange(min, max);
    let newPrice = item.basePrice * (1 + fluctuation);
    
    newPrice = Math.max(item.basePrice * 0.1, newPrice);
    newPrice = Math.min(item.basePrice * 3.0, newPrice);
    
    return Math.round(newPrice);
}

function calculateDailyPrice(item, news) {
    let price = item.currentPrice || item.basePrice;
    
    const naturalFluctuation = calculateNaturalFluctuation(item);
    price = naturalFluctuation;
    
    if (news && news.affectedItems.includes(item.id)) {
        price *= (1 + news.priceChange);
    }
    
    price = Math.max(item.basePrice * 0.1, price);
    price = Math.min(item.basePrice * 3.0, price);
    
    return Math.round(price);
}
```

### 3.2 生病概率算法

```javascript
function calculateSicknessChance(health) {
    if (health >= 90) return { chance: 0, type: null };
    if (health >= 70) return { chance: 0.05, type: 'minor' };
```

### 3.3 精力上限计算

```javascript
function calculateMaxEnergy(baseMax, health, consecutiveGymDays) {
    let maxEnergy = baseMax;
    
    if (health < 50) maxEnergy -= 2;
    if (health < 30) maxEnergy -= 2;
    
    if (consecutiveGymDays >= 3) maxEnergy += 1;
    if (consecutiveGymDays >= 7) maxEnergy += 1;
    
    return Math.max(1, Math.min(15, maxEnergy));
}
```

### 3.5 每日消费扣除算法

```javascript
function calculateDailyExpense(playerState) {
    let expense = 30;
    
    if (playerState.housingType === 'suburban') {
        expense += 50;
    } else if (playerState.housingType === 'urban') {
        expense += 100;
    } else if (playerState.housingType === 'owned') {
        expense += 20;
    }
    
    return expense;
}

function deductDailyExpense(playerState) {
    const expense = calculateDailyExpense(playerState);
    
    if (playerState.money >= expense) {
        playerState.money -= expense;
    } else {
        const deficit = expense - playerState.money;
        playerState.money = 0;
        playerState.privateLoan += deficit * 1.05;
        playerState.overdueDays++;
    }
}
```

### 3.7 职场变动算法

```javascript
function checkJobChange(playerState, workedToday, workedFullDay) {
    let promotionChance = 0.05;
    
    if (playerState.health >= 80 && workedFullDay) {
        promotionChance += 0.15;
    }
    if (playerState.reputation >= 70) {
        promotionChance += 0.05;
    }
    
    if (Math.random() < promotionChance && playerState.jobLevel < 5) {
        playerState.jobLevel++;
        return 'promotion';
    }
    
    if (Math.random() < 0.04 && !playerState.salaryDeduction) {
        playerState.salaryDeduction = true;
        playerState.salaryDeductionDays = 7;
        return 'salary_cut';
    }
    
    if (Math.random() < 0.03) {
        playerState.unemployed = true;
        playerState.unemployedDays = 3;
        return 'fired';
    }
    
    return null;
}
```

### 3.8 结局判定算法

```javascript
function determineEnding(playerState) {
    if (playerState.health <= 0) return ENDINGS.HEALTH_COLLAPSE;
    if (playerState.reputation < 0) return ENDINGS.REPUTATION_DESTROYED;
    if (checkBankruptcy(playerState)) return ENDINGS.BANKRUPTCY;
    
    if (playerState.day > 180 && !playerState.purchasedHouse) {
        return ENDINGS.TIME_UP;
    }
    
    if (playerState.purchasedHouse) {
        if (playerState.bankLoan > 0 || playerState.privateLoan > 0) {
            return ENDINGS.LOAN_DEBT;
        }
        if (playerState.reputation >= 20 && playerState.health >= 40) {
            return ENDINGS.RESENTFUL_MARRIAGE;
        }
        return ENDINGS.REGRET_BREAKUP;
    }
    
    return null;
}
```

## 四、系统流程规格

### 4.1 每日主循环流程

```
┌─────────────────────────────────────────────────────────┐
│                    每日主循环（市场核心版）              │
├─────────────────────────────────────────────────────────┤
│  1. 天数 + 1                                            │
│                                                         │
│  2. 自动扣除固定生活消费                                │
│     ├─ 吃饭+交通: 30金币                                │
│     ├─ 住房: 郊区50/市区100/已买房20                    │
│     └─ 金币不足 → 自动借入私人借贷 + 5%罚息             │
│                                                         │
│  3. 负债罚息结算                                        │
│     └─ 逾期负债每日扣除5%作为罚息                       │
│                                                         │
│  4. 刷新今日报纸（核心！）                              │
│     └─ 直接决定明日市场价格                             │
```

### 6.2 存储API封装

```javascript
const storage = {
    save(key, data) {
        try {
            wx.setStorageSync(key, JSON.stringify(data));
            return true;
        } catch (e) {
            console.error('存档失败', e);
            return false;
        }
    },
    
    load(key) {
        try {
            const data = wx.getStorageSync(key);
            return data ? JSON.parse(data) : null;
        } catch (e) {
            console.error('读档失败', e);
            return null;
        }
    },
    
    remove(key) {
        try {
            wx.removeStorageSync(key);
            return true;
        } catch (e) {
            return false;
        }
    }
};
```
